def initialize(socket, ctx)
  @socket = socket
  @ctx = ctx
  @eng_ctx = IS_JRUBY ? @ctx : SSLContext.new(ctx)
end

def accept
  @ctx.check
  io = @socket.accept
  engine = Engine.server @eng_ctx
  Socket.new io, engine
end
