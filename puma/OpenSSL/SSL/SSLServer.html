def accept_nonblock
  sock = @svr.accept_nonblock

  begin
    ssl = OpenSSL::SSL::SSLSocket.new(sock, @ctx)
    ssl.sync_close = true
    ssl.accept if @start_immediately
    ssl
  rescue SSLError => ex
    if ssl
      ssl.close
    else
      sock.close
    end
    raise ex
  end
end
