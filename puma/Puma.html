def self.abstract_unix_socket?
  @abstract_unix ||=
    if HAS_UNIX_SOCKET
      begin
        ::UNIXServer.new("\0puma.temp.unix").close
        true
      rescue ArgumentError  # darwin
        false
      end
    else
      false
    end
end
